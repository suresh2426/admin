	MainActivity.java
package com.example.app1;

import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.TextView;
import android.widget.Toast;

import androidx.activity.EdgeToEdge;
import androidx.appcompat.app.AppCompatActivity;
import androidx.core.graphics.Insets;
import androidx.core.view.ViewCompat;
import androidx.core.view.WindowInsetsCompat;

public class MainActivity extends AppCompatActivity {

   TextView t;
   Button btn;
   EditText msg, num1, num2;

   @Override
   protected void onCreate(Bundle savedInstanceState) {
       super.onCreate(savedInstanceState);
       EdgeToEdge.enable(this);
       setContentView(R.layout.activity_main);
       ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main), (v, insets) -> {
           Insets systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars());
           v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom);
           return insets;
       });

       t = findViewById(R.id.txtbox);
       btn = findViewById(R.id.button);
       msg = findViewById(R.id.msg);
       num1 = findViewById(R.id.num1);
       num2 = findViewById(R.id.num2);
   }

   public void show(View view) {
       String operation = msg.getText().toString().trim();
       try {
           int a = Integer.parseInt(num1.getText().toString());
           int b = Integer.parseInt(num2.getText().toString());

           if (operation.equalsIgnoreCase("Add")) {
               t.setText("Addition: " + (a + b));
           } else if (operation.equalsIgnoreCase("Subtract")) {
               t.setText("Subtraction: " + (a - b));
           } else if (operation.equalsIgnoreCase("Multiply")) {
               t.setText("Multiplication: " + (a * b));
           } else if (operation.equalsIgnoreCase("Divide")) {
               if (b != 0)
                   t.setText("Division: " + (a / b));
               else
                   t.setText("Cannot divide by zero");
           } else {
               t.setText("Invalid operation");
           }

           Toast.makeText(getApplicationContext(), "Operation performed!", Toast.LENGTH_LONG).show();

       } catch (NumberFormatException e) {
           t.setText("Please enter valid numbers");
       }
   }
}

package com.example.app1;

import android.content.Intent;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;

import androidx.activity.EdgeToEdge;
import androidx.appcompat.app.AppCompatActivity;
import androidx.core.graphics.Insets;
import androidx.core.view.ViewCompat;
import androidx.core.view.WindowInsetsCompat;

public class MainActivity extends AppCompatActivity {

   EditText idInput, nameInput, emailInput;
   Button next;

   @Override
   protected void onCreate(Bundle savedInstanceState) {
       super.onCreate(savedInstanceState);
       EdgeToEdge.enable(this);
       setContentView(R.layout.activity_main);
       ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main), (v, insets) -> {
           Insets systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars());
           v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom);
           return insets;
       });

       idInput = findViewById(R.id.editId);
       nameInput = findViewById(R.id.editName);
       emailInput = findViewById(R.id.editEmail);
       next = findViewById(R.id.next);

       next.setOnClickListener(new View.OnClickListener() {
           @Override
           public void onClick(View v) {
               int id = Integer.parseInt(idInput.getText().toString());
               String name = nameInput.getText().toString();
               String email = emailInput.getText().toString();

               Intent nextActivity = new Intent(MainActivity.this, MainActivity2.class);
               nextActivity.putExtra("id", id);
               nextActivity.putExtra("name", name);
               nextActivity.putExtra("email", email);
               startActivity(nextActivity);
           }
       });
   }
}

   protected void onCreate(Bundle savedInstanceState) {
       super.onCreate(savedInstanceState);

package com.example.app1;

import android.content.Intent;
import android.os.Bundle;
import android.view.View;
import android.widget.EditText;
import android.widget.ImageButton;

import androidx.activity.EdgeToEdge;
import androidx.appcompat.app.AppCompatActivity;
import androidx.core.graphics.Insets;
import androidx.core.view.ViewCompat;
import androidx.core.view.WindowInsetsCompat;

public class MainActivity extends AppCompatActivity {

   EditText to, cc, subject, body;
   ImageButton send;

   @Override
   protected void onCreate(Bundle savedInstanceState) {
       super.onCreate(savedInstanceState);
       EdgeToEdge.enable(this);
       setContentView(R.layout.activity_main);
       ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main), (v, insets) -> {
           Insets systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars());
           v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom);
           return insets;
       });

       cc = findViewById(R.id.cc);
       to = findViewById(R.id.to);
       subject = findViewById(R.id.subject);
       body = findViewById(R.id.body);
       send = findViewById(R.id.send);

       send.setOnClickListener(v -> {
           Intent mail = new Intent(Intent.ACTION_SEND);
           mail.setType("message/rfc822");
           mail.putExtra(Intent.EXTRA_EMAIL, new String[]{to.getText().toString()});
           mail.putExtra(Intent.EXTRA_CC, new String[]{cc.getText().toString()});
           mail.putExtra(Intent.EXTRA_SUBJECT, subject.getText().toString());
           mail.putExtra(Intent.EXTRA_TEXT, body.getText().toString());
           startActivity(Intent.createChooser(mail, "Select your email app:"));
       });
   }
}

package com.example.sqlitedatabase;

import android.content.ContentValues;
import android.content.Context;
import android.database.Cursor;
import android.database.sqlite.SQLiteDatabase;
import android.database.sqlite.SQLiteOpenHelper;

import androidx.annotation.Nullable;

public class DatabaseHelper extends SQLiteOpenHelper {

   SQLiteDatabase sqLiteDatabase;

   public DatabaseHelper(@Nullable Context context) {
       super(context, "MCA202426", null, 1);
       sqLiteDatabase = getWritableDatabase();
   }

   @Override
   public void onCreate(SQLiteDatabase db) {
       db.execSQL("CREATE TABLE IF NOT EXISTS MCA(rollno INTEGER PRIMARY KEY AUTOINCREMENT, email TEXT, password TEXT, usertype TEXT)");
   }

   @Override
   public void onUpgrade(SQLiteDatabase db, int oldVersion, int newVersion) {}

   public boolean insertRecord(String email, String password, String userType) {
       ContentValues cv = new ContentValues();
       cv.put("email", email);
       cv.put("password", password);
       cv.put("usertype", userType);
       long result = sqLiteDatabase.insert("MCA", null, cv);
       return result != -1;
   }

   public Cursor showRecords() {
       return sqLiteDatabase.rawQuery("SELECT * FROM MCA", null);
   }

   public Cursor searchRecords(int roll) {
       sqLiteDatabase = getReadableDatabase();
       return sqLiteDatabase.rawQuery("SELECT * FROM MCA WHERE rollno=" + roll, null);
   }

   public boolean updateRecords(int roll_new, String email, String password, String userType) {
       ContentValues cv = new ContentValues();
       cv.put("email", email);
       cv.put("password", password);
       cv.put("usertype", userType);
       long result = sqLiteDatabase.update("MCA", cv, "rollno=?", new String[]{String.valueOf(roll_new)});
       return result != -1;
   }

   public boolean deleteRecords(int roll_del) {
       long result = sqLiteDatabase.delete("MCA", "rollno=?", new String[]{String.valueOf(roll_del)});
       return result != -1;
   }
}

       ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main), (v, insets) -> {
           Insets systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars());
           v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom);
           return insets;
       });

       email = findViewById(R.id.email);
       password = findViewById(R.id.password);
       tv = findViewById(R.id.tv);
       info = findViewById(R.id.info);
       login = findViewById(R.id.login);
       builder = new AlertDialog.Builder(this);

       tv.setOnClickListener(new View.OnClickListener() {
           @Override
           public void onClick(View v) {
               String languages[] = {"English", "मराठी", "हिन्दी", "ગુજરાતી", "اردو"};

               builder.setTitle(getString(R.string.change_language));
               builder.setSingleChoiceItems(languages, -1, new DialogInterface.OnClickListener() {
                   @Override
                   public void onClick(DialogInterface dialog, int which) {
                       String lang = languages[which];

                       if (lang.equals("English")) {
                           setLocale("");
                       } else if (lang.equals("मराठी")) {
                           setLocale("mr");
                       } else if (lang.equals("हिन्दी")) {
                           setLocale("hi"); // Corrected from "hr"
                       } else if (lang.equals("ગુજરાતી")) {
                           setLocale("gu");
                       } else if (lang.equals("اردو")) {
                           setLocale("ur");
                       }

                       recreate();
                       Toast.makeText(getApplicationContext(), "You selected: " + lang, Toast.LENGTH_LONG).show();
                   }
               });
               builder.create();
               builder.show();
           }
       });
   }

   private void setLocale(String lang) {
       Locale locale = new Locale(lang);
       Locale.setDefault(locale);
       Configuration configuration = new Configuration();
       configuration.setLocale(locale);
       getBaseContext().getResources().updateConfiguration(configuration, getResources().getDisplayMetrics());

       SharedPreferences.Editor editor = getSharedPreferences("Settings", MODE_PRIVATE).edit();
       editor.putString("Language", lang);
       editor.apply();
   }

   private void loadLocale() {
       SharedPreferences sharedPreferences = getSharedPreferences("Settings", MODE_PRIVATE);
       String lang = sharedPreferences.getString("Language", "");
       setLocale(lang);
   }
}

package com.example.firebaseproj;
 
public class Student {
 
	public String id;
	public String name;
	public String address;
	public String course;
 
	public Student(){}
 
	public Student(String id, String name, String address, String course){
    	this.id = id;
    	this.name = name;
    	this.address = address;
    	this.course = course;
	}
 
	public String getId() {
    	return id;
	}
 
	public void setId(String id) {
    	this.id = id;
	}
 
	public String getName() {
    	return name;
	}
 
	public void setName(String name) {
    	this.name = name;
	}
 
	public String getAddress() {
    	return address;
	}
 
	public void setAddress(String address) {
    	this.address = address;
	}
 
	public String getCourse() {
    	return course;
	}
 
	public void setCourse(String course) {
    	this.course = course;
	}
}
